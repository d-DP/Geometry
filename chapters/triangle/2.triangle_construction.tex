\begin{frame}
Given a=8cm, c-b=k (k=3.5cm)
Apply cosine rule 
\begin{align*}
	\cos{(B)} = \frac{a^2+c^2-b^2}{2ac}\\
	\cos{(B)}= \frac{a^2+(b+k)^2-b^2}{2a(b+k)}\\
	2ab\cos{B}+2ak\cos{B}=a^2+k+2bk\\
	b=\frac{a^2 + k^2-2ak \cos{B} }{2a\cos{B}-2k}
\end{align*}
\begin{center}
b=8.49, c=11.99
\end{center}
